## Install

```bash
npm install --save react-data-table-modify
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-data-table-modify'
import 'react-data-table-modify/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
